<script>
let linksData = null;

function loadSecret(){
  const saved = localStorage.getItem('u24_admin_secret');
  if(saved){ document.getElementById('adminSecret').value = saved; }
}
document.getElementById('adminSecret').addEventListener('change', e=>{
  localStorage.setItem('u24_admin_secret', e.target.value);
});

async function init(){
  loadSecret();
  try{
    const res = await fetch('links.json?v='+Date.now());
    linksData = await res.json();
    document.getElementById('loadingMsg').style.display='none';
    document.getElementById('mainContent').style.display='block';
    renderHome();
    renderGuides();
    document.getElementById('saveBtn').disabled = false;
  }catch(err){
    document.getElementById('loadingMsg').innerText = 'links.json load করতে সমস্যা হয়েছে: ' + err.message;
  }
}

function renderHome(){
  const panel = document.getElementById('panel-home');
  panel.innerHTML = '';
  linksData.homepage_services.forEach((cat, ci)=>{
    const div = document.createElement('div');
    div.className = 'cat';
    div.innerHTML = `
      <div class="cat-head" onclick="toggleCat(${ci})">
        <h3>${cat.icon || '📁'} ${cat.title || '(no title)'} — <span style="color:#7a8699;font-weight:600">${cat.items.length} links</span></h3>
        <span>▾</span>
      </div>
      <div class="cat-body" id="catbody-${ci}">
        <div class="cat-meta">
          <input value="${escapeAttr(cat.icon)}" placeholder="Icon" onchange="updateCatMeta(${ci},'icon',this.value)">
          <input value="${escapeAttr(cat.title)}" placeholder="Category title" onchange="updateCatMeta(${ci},'title',this.value)">
          <input value="${escapeAttr(cat.desc)}" placeholder="Description" onchange="updateCatMeta(${ci},'desc',this.value)">
        </div>
        <div class="row-labels"><span>Label</span><span>Description</span><span>URL</span><span>Status</span><span></span></div>
        <div id="items-${ci}">
          ${cat.items.map((it,ii)=>itemRowHTML(ci,ii,it)).join('')}
        </div>
        <button class="addItemBtn" onclick="addItem(${ci})">+ Add new link to this category</button>
        <div style="margin-top:10px;text-align:right">
          <button class="delCatBtn" onclick="deleteCat(${ci})">🗑️ Delete entire category</button>
        </div>
      </div>
    `;
    panel.appendChild(div);
  });
  const addCatBtn = document.createElement('button');
  addCatBtn.className = 'addCatBtn';
  addCatBtn.innerText = '+ Add New Category';
  addCatBtn.onclick = addCategory;
  panel.appendChild(addCatBtn);
}

function itemRowHTML(ci, ii, it){
  const statuses = ['Online','Updating','Call'];
  return `
    <div class="item-row" id="item-${ci}-${ii}">
      <input value="${escapeAttr(it[0])}" placeholder="Label" onchange="updateItem(${ci},${ii},0,this.value)">
      <input value="${escapeAttr(it[1])}" placeholder="Description" onchange="updateItem(${ci},${ii},1,this.value)">
      <input value="${escapeAttr(it[3])}" placeholder="URL" onchange="updateItem(${ci},${ii},3,this.value)">
      <select onchange="updateItem(${ci},${ii},2,this.value)">
        ${statuses.map(s=>`<option value="${s}" ${it[2]===s?'selected':''}>${s}</option>`).join('')}
      </select>
      <button onclick="deleteItem(${ci},${ii})">Delete</button>
    </div>
  `;
}

function escapeAttr(s){
  if(s===undefined||s===null) return '';
  return String(s).replace(/&/g,'&amp;').replace(/"/g,'&quot;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function toggleCat(ci){
  document.getElementById('catbody-'+ci).classList.toggle('open');
}

function updateCatMeta(ci, field, val){
  linksData.homepage_services[ci][field] = val;
  renderHome_preserveOpen();
}

function updateItem(ci, ii, idx, val){
  linksData.homepage_services[ci].items[ii][idx] = val;
}

function deleteItem(ci, ii){
  if(!confirm('এই link delete করতে চান?')) return;
  linksData.homepage_services[ci].items.splice(ii,1);
  renderHome_preserveOpen(ci);
}

function addItem(ci){
  linksData.homepage_services[ci].items.push(['New Link','Description','Online','https://']);
  renderHome_preserveOpen(ci);
}

function deleteCat(ci){
  if(!confirm('পুরো category delete করতে চান? এর ভেতরের সব link চলে যাবে।')) return;
  linksData.homepage_services.splice(ci,1);
  renderHome();
}

function addCategory(){
  linksData.homepage_services.push({
    key: 'new category keywords here',
    title: 'New Category',
    icon: '📁',
    desc: 'Description here',
    items: [['New Link','Description','Online','https://']]
  });
  renderHome();
  setTimeout(()=>{ document.getElementById('catbody-'+(linksData.homepage_services.length-1)).classList.add('open'); }, 50);
}

let openCats = new Set();
function renderHome_preserveOpen(keepOpen){
  document.querySelectorAll('.cat-body.open').forEach(el=>{
    const idx = el.id.replace('catbody-','');
    openCats.add(idx);
  });
  renderHome();
  openCats.forEach(idx=>{
    const el = document.getElementById('catbody-'+idx);
    if(el) el.classList.add('open');
  });
  if(keepOpen!==undefined){
    const el = document.getElementById('catbody-'+keepOpen);
    if(el) el.classList.add('open');
  }
}

function renderGuides(){
  const panel = document.getElementById('panel-guides');
  panel.innerHTML = '';
  Object.keys(linksData.guide_pages).forEach(page=>{
    const btns = linksData.guide_pages[page];
    const div = document.createElement('div');
    div.className = 'guide-page-block';
    div.innerHTML = `
      <h3>📄 ${page}</h3>
      <div class="row-labels"><span>Label</span><span>Description</span><span>URL</span><span>Class</span><span></span></div>
      <div id="guide-items-${page}">
        ${btns.map((b,bi)=>guideRowHTML(page,bi,b)).join('')}
      </div>
      <button class="addItemBtn" onclick="addGuideBtn('${page}')">+ Add new button to this page</button>
    `;
    panel.appendChild(div);
  });
}

function guideRowHTML(page, bi, b){
  return `
    <div class="guide-row" id="guide-${page}-${bi}">
      <input value="${escapeAttr(b.label)}" placeholder="Label" onchange="updateGuideBtn('${page}',${bi},'label',this.value)">
      <input value="${escapeAttr(b.url)}" placeholder="URL" onchange="updateGuideBtn('${page}',${bi},'url',this.value)" style="grid-column:span 2">
      <input value="${escapeAttr(b.class)}" placeholder="CSS class (orange/green/blank)" onchange="updateGuideBtn('${page}',${bi},'class',this.value)">
      <button onclick="deleteGuideBtn('${page}',${bi})">Delete</button>
    </div>
  `;
}

function updateGuideBtn(page, bi, field, val){
  linksData.guide_pages[page][bi][field] = val;
}

function deleteGuideBtn(page, bi){
  if(!confirm('এই button delete করতে চান?')) return;
  linksData.guide_pages[page].splice(bi,1);
  renderGuides();
}

function addGuideBtn(page){
  linksData.guide_pages[page].push({label:'New Button', url:'https://', class:''});
  renderGuides();
}

document.querySelectorAll('.tab').forEach(tab=>{
  tab.addEventListener('click', ()=>{
    document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
    document.querySelectorAll('.panel').forEach(p=>p.classList.remove('active'));
    tab.classList.add('active');
    document.getElementById('panel-'+tab.dataset.tab).classList.add('active');
  });
});

document.getElementById('saveBtn').addEventListener('click', async ()=>{
  const secret = document.getElementById('adminSecret').value.trim();
  if(!secret){
    setStatus('err', 'Admin secret key দিন আগে।');
    return;
  }
  setStatus('busy', 'Saving... GitHub-এ commit হচ্ছে...');
  document.getElementById('saveBtn').disabled = true;
  try{
    const res = await fetch('/api/update-links', {
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body: JSON.stringify({ secret, data: linksData })
    });
    const result = await res.json();
    if(res.ok && result.success){
      setStatus('ok', '✅ Saved! ১-২ মিনিটে live site update হয়ে যাবে।');
    }else{
      setStatus('err', '❌ Error: ' + (result.error || 'Unknown error'));
    }
  }catch(err){
    setStatus('err', '❌ Network error: ' + err.message);
  }
  document.getElementById('saveBtn').disabled = false;
});

function setStatus(type, msg){
  const el = document.getElementById('statusMsg');
  el.className = 'status ' + type;
  el.innerText = msg;
}

init();
</script>
